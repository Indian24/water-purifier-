// client/src/pages/Contact.tsx
import React, { useState } from "react";

export default function ContactPage() {
  const [form, setForm] = useState({ name: "", phone: "", email: "", message: "" });
  const [status, setStatus] = useState<{ ok: boolean; msg: string } | null>(null);

  const submit = async (e: React.FormEvent) => {
    e.preventDefault();
    setStatus(null);
    try {
      const res = await fetch("/api/contact", {
        method: "POST",
        headers: { "content-type": "application/json" },
        body: JSON.stringify(form),
      });
      if (!res.ok) throw new Error(await res.text());
      setStatus({ ok: true, msg: "Thanks — we received your message." });
      setForm({ name: "", phone: "", email: "", message: "" });
    } catch (err: any) {
      setStatus({ ok: false, msg: err.message || "Failed to send" });
    }
  };

  return (
    <div className="max-w-4xl mx-auto px-4 py-12">
      <h1 className="text-3xl font-bold mb-4">Contact & Support</h1>
      <p className="mb-8 text-gray-600">For repair, installation and orders — reach out below.</p>

      <div className="grid grid-cols-1 md:grid-cols-2 gap-6 mb-8">
        <div className="p-6 rounded-lg shadow-sm bg-white">
          <h3 className="font-semibold">Ranjeet Sharma</h3>
          <p className="text-sm text-gray-600">Senior Technician</p>
          <a className="text-blue-600 mt-2 inline-block" href="tel:+919910472820">+91 99104 72820</a>
        </div>
        <div className="p-6 rounded-lg shadow-sm bg-white">
          <h3 className="font-semibold">Manish Ranjan</h3>
          <p className="text-sm text-gray-600">Orders & Enquiries</p>
          <a className="text-blue-600 mt-2 inline-block" href="tel:+918700762477">+91 87007 62477</a>
        </div>
      </div>

      <div className="mb-8">
        <p><strong>Email:</strong> <a href="mailto:ranjanmanish615@gmail.com">ranjanmanish615@gmail.com</a></p>
        <p className="mt-2"><strong>Office:</strong> House No. 40, Rajpur Khurd, Chhatarpur, South Delhi - 110068</p>
      </div>

      <form onSubmit={submit} className="space-y-4 bg-white p-6 rounded-md shadow-sm">
        <input className="w-full p-3 border rounded" placeholder="Your name" value={form.name} onChange={e => setForm({...form, name: e.target.value})} required />
        <input className="w-full p-3 border rounded" placeholder="Phone" value={form.phone} onChange={e => setForm({...form, phone: e.target.value})} required />
        <input className="w-full p-3 border rounded" placeholder="Email" type="email" value={form.email} onChange={e => setForm({...form, email: e.target.value})} />
        <textarea className="w-full p-3 border rounded" placeholder="Message" value={form.message} onChange={e => setForm({...form, message: e.target.value})} />
        <div className="flex items-center gap-4">
          <button className="px-6 py-2 bg-blue-600 text-white rounded">Send Message</button>
          {status && <span className={status.ok ? "text-green-600" : "text-red-600"}>{status.msg}</span>}
        </div>
      </form>
    </div>
  );
}
